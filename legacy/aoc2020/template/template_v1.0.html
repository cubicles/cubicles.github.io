def day1part1(input):
    for i in range(len(input)):
        for j in range(i+1,len(input)):
            sum = input[i]+input[j]
            if sum == 2020:
                answer = input[i]*input[j]
    return answer

#Second part
def day1part2(input):
    for i in range(len(input)):
        for j in range(i+1,len(input)):
            for k in range(i+2,len(input)):
                sum = input[i]+input[j]+input[k]
                if sum == 2020:
                    answer = input[i]*input[j]*input[k]
    return answer
